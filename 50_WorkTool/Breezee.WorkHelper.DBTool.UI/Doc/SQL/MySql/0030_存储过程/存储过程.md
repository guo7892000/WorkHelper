/*创建函数存储过程1*/
CREATE DEFINER=`mysqladmin`@`%` PROCEDURE `test`.`pr_sys_deltemp`(IN p_itmid varchar(30),
OUT OSA_RETURN_CODE varchar(5),
OUT OSA_SQLERRM varchar(500))
__END: BEGIN

     declare ls_date varchar(8);
      /*----------      异常处理机制  模板     ----------*/
  DECLARE EXIT HANDLER FOR SQLWARNING,NOT FOUND,SQLEXCEPTION -- 如果有了游标，就不能有 NOT FOUND
  BEGIN
   GET DIAGNOSTICS CONDITION 1
   OSA_RETURN_CODE = RETURNED_SQLSTATE, OSA_SQLERRM = MESSAGE_TEXT;
  END;
     set OSA_RETURN_CODE ="OK";
     set OSA_SQLERRM="noErr";
     /*-------------     开始业务逻辑     --------------*/

     -- 根据 通知书获取 需要转换的 office 文件ID
     -- 清除部品临时表
     delete from t_test_material_temp a ;
     -- 清除展开信息表
     delete from T_test_msg_temp a ;
     -- 清除展开临时表
     delete from T_test_ins_temp a ;
END
