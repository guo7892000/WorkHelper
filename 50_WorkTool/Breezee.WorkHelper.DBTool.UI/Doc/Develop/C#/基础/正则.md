### 示例1
```
String sUpdateSetPattern = "^UPDATE\\s*\\S*\\s*SET\\s*";	
Regex regex = new Regex(sUpdateSetPattern, RegexOptions.IgnoreCase);	
MatchCollection mc = regex.Matches(sSql);	
foreach (Match m in mc)	
{	
    sb.Append(m.Value);//不变的UPDATE SET部分先加入	
    sSql = sSql.Substring(m.Index + m.Value.Length).Trim();	
    //调用From方法	
    sb.Append(fromSqlConvert(sSql));	
}	
```
