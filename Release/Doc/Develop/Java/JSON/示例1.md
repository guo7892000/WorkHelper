## JAVA-JSON示例1
### 示例1（org.codehaus.jettison）
```
public void ReadTimeSeriesJson(String authorization,String json) throws Exception {         		
	org.codehaus.jettison.json.JSONArray array = new org.codehaus.jettison.json.JSONArray(json);//转换JSON字符为JSON数组		
	String stagName="name",sDatapointsName="datapoints",sAttributesName="attributes";		
	for (int i = 0; i < array.length(); i++) {		
		org.codehaus.jettison.json.JSONArray arrDatapoints = null;		
		org.codehaus.jettison.json.JSONObject arrAttributes= null;		
		org.codehaus.jettison.json.JSONObject jsonObject = array.getJSONObject(i);//当前对象		
		if(jsonObject.has(stagName) && "".equalsIgnoreCase(jsonObject.getString(stagName))== false) {		
			System.out.println(jsonObject.getString(stagName)); //"Compressor-2015:CompressionRatio"		
		}		
		if(jsonObject.has(sDatapointsName)) { 		
			arrDatapoints = jsonObject.getJSONArray(sDatapointsName); //[[1453338377222,10,1],[1453338377222,10,1]]		
		}		
		if(jsonObject.has(sAttributesName)) {		
			arrAttributes = jsonObject.getJSONObject(sAttributesName);		
		}		
		//时间戳数据处理：[[1489615262095,2.757900075726165,3],[1489615293324,2.7387557351287115,3]]		
		for (int j = 0; j < arrDatapoints.length(); j++) { 		
			List<Object> datapoint1 = new ArrayList<Object>();		
			org.codehaus.jettison.json.JSONArray jsonData = arrDatapoints.getJSONArray(j);//[1453338377222,10,1]		
			System.out.println("jsonData:"+jsonData.toString());		
			for (int k = 0; k < jsonData.length(); k++) {		
				datapoint1.add(jsonData.get(k));		
			}		
		}		
		//属性处理:{"test":"propertytest","cust":"zhangsan"}		
		com.fasterxml.jackson.databind.ObjectMapper mapper = new com.fasterxml.jackson.databind.ObjectMapper();		
		Map<String,Object> m = mapper.readValue(arrAttributes.toString(), Map.class);//将属性转换为Map对象		
		for (java.util.Map.Entry<String, Object> key :m.entrySet()) {   		
			System.out.println(key.getKey()+":"+key.getValue());		
		}		
	}		
}		

```
