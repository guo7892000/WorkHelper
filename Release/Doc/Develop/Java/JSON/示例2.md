## JAVA-JSON示例2
### 示例2（org.codehaus.jettison）
```
import org.springframework.web.client.RestTemplate;
import org.springframework.http.converter.StringHttpMessageConverter;

RestTemplate restTemplate = new RestTemplate();
restTemplate.getMessageConverters().add(new StringHttpMessageConverter());
String resString = restTemplate.getForObject(sbUrl.toString(),String.class);
// logger.debug("返回字符实体内容：" + resString);
// 读取JSON对象
org.codehaus.jettison.json.JSONObject objMain = new org.codehaus.jettison.json.JSONObject(resString);
DlrPartQtyResponse ponse= new  DlrPartQtyResponse();
ponse.setCode(objMain.getInt("code"));
ponse.setCount(objMain.getInt("count"));
List<DlrPartQty> lst = new ArrayList<DlrPartQty>();
            
org.codehaus.jettison.json.JSONArray arrData= objMain.getJSONArray("data");
for(int i=0;i<arrData.length();i++){
  org.codehaus.jettison.json.JSONObject jsonOne = arrData.getJSONObject(i);//当前对象
  DlrPartQty qty = new DlrPartQty();
  qty.setDlrId(jsonOne.getString("dlrId"));
  qty.setPartNo(jsonOne.getString("partNo"));
  qty.setQty(jsonOne.getString("qty"));
  lst.add(qty);
 }
ponse.setData(lst);
```
引用JAR包：  
```
<!-- https://mvnrepository.com/artifact/org.codehaus.jettison/jettison -->	
<dependency>	
    <groupId>org.codehaus.jettison</groupId>	
    <artifactId>jettison</artifactId>	
    <version>1.4.0</version>	
</dependency>	
```
