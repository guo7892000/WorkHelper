### 实例-接收接口格式
```
/*删除存储过程*/
IF ( OBJECT_ID('IFR_GD_T_CAR_INFO') IS NOT NULL ) 
	DROP PROCEDURE IFR_GD_T_CAR_INFO
GO	
/*创建存储过程*/
CREATE PROCEDURE IFR_GD_T_CAR_INFO 
AS
/***********************************************************************************
* 存储过程名称: xxx信息接收
* 创建作者: 张三
* 创建日期: 2012-12-05   
* 使用系统: DMS
* 使用模块: xxx模块
* 说明：xxx信息接收
* 修改历史说明：
*	2012-12-05(张三): 由CLR存储过程转换为SQL存储过程(CLR布署太麻烦了)   
***********************************************************************************/ 
BEGIN
	DECLARE @curTemp CURSOR
	SET @curTemp = Cursor For SELECT In_Id 
		FROM  AA		  
	OPEN @curTemp
	FETCH NEXT FROM @curTemp INTO @In_Id 		  
	WHILE @@fetch_status = 0 
		BEGIN		
					  
			FETCH NEXT FROM @curTemp INTO @In_Id
		END 	
	close @curTemp
	deallocate @curTemp 	
END

```
