CREATE OR REPLACE Package Body Package PKG_TEST_01
Is
 Procedure P_Item_One(i_operid        Varchar2,
                                i_itemid        Varchar2, --
                                o_Return_Code   Out Varchar2, --返回代码
                                o_Error_Msg Out Varchar2, --业务错误信息
                                o_Sqlerrm       Out Varchar2 --存储过程异常
                                ) Is
 
   var_str     Varchar2(6);
   var_teststr Clob;
   var_specstr Varchar2(2000);
   var_md5     Varchar2(32);
   var_count   Integer;
   Row_config  t_test_config%Rowtype;
 Begin
   o_Return_Code := pkg_com_define.const_ok;
   Row_config := Null;
 
   --
   For r_up In (Select item_cid || '|' || qty As ebomstr
                  From t_test_stru a
                 Where a.item_pid = i_itemid
                 Order By a.item_cid, a.qty) Loop
     var_teststr := var_teststr || var_str || r_up.ebomstr;
     var_str     := ',';
   End Loop;
 
   --
   var_str := Null;
   For r_up In (Select a.spec_code
                  From t_item_spec a
                 Where a.item_id = i_itemid
                 Order By a.spec_code) Loop
     var_specstr := var_specstr || var_str || r_up.spec_code;
     var_str     := ',';
   End Loop;  
   --
   Begin
     var_md5 := utl_raw.cast_to_raw(dbms_obfuscation_toolkit.MD5(input_string => var_teststr));
     Row_config.con_id      := sys_guid();
     Row_config.item_id     := i_itemid;
     Row_config.creater     := i_operid;
     Row_config.create_time := Sysdate;
   
     Insert Into t_test_config Values Row_config;
   End;
 
 Exception
   When Others Then
     o_return_code := 'ERROR';
     o_sqlerrm     := Sqlerrm;
 End;

 Procedure P_Item_All(i_itemid        Varchar2, --
                               o_Return_Code   Out Varchar2, --返回代码
                               o_Error_Msg Out Varchar2, --业务错误信息
                               o_Sqlerrm       Out Varchar2 --存储过程异常
                               ) Is
 Begin
   o_Return_Code := pkg_com_define.const_ok;
 
   For R_UP In (Select *
                  From t_test_gc c
                 Where c.date_end = '9'
                   And num_from = 'M'
) Loop
     P_Item_One(i_operid        => R_UP.creater,
                          i_itemid        => R_UP.ITEM_ID, --
                          o_Return_Code   => o_Return_Code, --返回代码
                          o_Error_Message => o_Error_Message, --业务错误信息
                          o_Sqlerrm       => o_Sqlerrm --存储过程异常
                          );
     Commit;
   End Loop;
 
 Exception
   When Others Then
     o_return_code := 'ERROR';
     o_sqlerrm     := Sqlerrm;
 End;
End Package PKG_TEST_01;
